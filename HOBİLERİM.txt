        var map = L.map('map').setView([38.5, 38.0], 6);

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '© OpenStreetMap'
        }).addTo(map);

        // İstediğiniz şehirlerin listesi
        var locations = [
            { name: "Adıyaman (Görev Yerim)", coords: [37.7648, 38.2786] },
            { name: "Van", coords: [38.4891, 43.4089] },
            { name: "Hakkari", coords: [37.5744, 43.7408] },
            { name: "Mersin", coords: [36.8121, 34.6415] },
            { name: "Konya (Eğitim)", coords: [37.8714, 32.4921] },
            { name: "Nevşehir", coords: [38.6244, 34.7144] }
        ];

        locations.forEach(function(loc) {
            L.marker(loc.coords).addTo(map).bindPopup(loc.name);
        });
